mehrere docker compose files zu einem machen
docker-compose -f FILE1 -f FILE2 config > AUSGABEDATEI

docker builden
	mit folgenden parameter wird kein lokaler cache verwendet und alle verwendeten container werden online geholt (immer aktuell)
	buildHardOptions="--force-rm --no-cache --pull"
docker-compose -f AUSGABEDATEI build ${buildHardOptions} ${containers}


container starten
docker-compose -f AUSGABEDATEI up -d $containers

logs anzeigen
docker-compose -f AUSGABEDATEI logs -f $containers


function doit {
    case "$1" in
        "build")
            docker-compose -f AUSGABEDATEI build
            ;;
        "build_hard")
            docker-compose -f AUSGABEDATEI build --force-rm --no-cache --pull
            ;;
        "run")
            docker-compose -f AUSGABEDATEI up -d
            ;;
        "Quit")
            echo "exit"
            exit 0
            ;;
        *)
            echo "unknown option"
            ;;
    esac
}
